```python
#!/bin/python3
import turtle

tina = turtle.Turtle()
tina_color = input('What color should Tina be?\n')
tina.color(tina_color)
myscreen = turtle.Screen()
bg_color = input('What color do you want the background to be?\n')
myscreen.bgcolor(bg_color)
```
